<task type="auto">
  <name>Task 2: Implement Moderation Workflow</name>
  <files>backend/internal/modules/comment/service.go</files>
  <action>
    Implement the moderation logic (per D-07):
    1. `SubmitComment`: Always set status to `Pending` by default.
    2. `ModerateComment`: Allow admins to change status to `Approved` or `Rejected`.
    3. Ensure that public retrieval APIs only return comments with `status = 'Approved'`.
  </action>
  <verify>
    <automated>
      # Submit comment, verify it's not in the public list, approve it, verify it's now visible
      curl -X POST http://localhost:8080/api/v3/comments -d '{"post_id": "...", "content": "Hi"}' && \
      curl -X GET http://localhost:8080/api/v3/posts/.../comments | grep -v "Hi" && \
      curl -X PATCH http://localhost:8080/api/v3/comments/.../status -d '{"status": "Approved"}' && \
      curl -X GET http://localhost:8080/api/v3/posts/.../comments | grep "Hi"
    </automated>
  </verify>
  <done>Comments are moderated and only approved ones are visible</done>
</task>
